        let currentFilteredList = [...bibleData];

        let currentIndex = 0;

        let currentMode = 'origin'; // origin, partial, random, all


        // 초기 실행

        window.onload = function() {

            renderCard();

            

            // 키보드 방향키 바인딩

            document.addEventListener('keydown', function(e) {

                if(e.key === 'ArrowLeft') prevVerse();

                if(e.key === 'ArrowRight') nextVerse();

            });

        };


        // 카테고리 필터 전환 (전체, 구약, 신약)

        function filterCategory(cat) {

            document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));

            if(cat === 'all') {

                currentFilteredList = [...bibleData];

                document.querySelectorAll('.tab-btn')[0].classList.add('active');

            } else if (cat === 'OT') {

                currentFilteredList = bibleData.filter(v => v.cat === 'OT');

                document.querySelectorAll('.tab-btn')[1].classList.add('active');

            } else if (cat === 'NT') {

                currentFilteredList = bibleData.filter(v => v.cat === 'NT');

                document.querySelectorAll('.tab-btn')[2].classList.add('active');

            }

            currentIndex = 0;

            renderCard();

        }


        // 가리기 모드 변경

        function changeMode(mode) {

            currentMode = mode;

            document.querySelectorAll('.mode-btn').forEach(btn => btn.classList.remove('active'));

            

            const btnIdx = { 'origin': 0, 'partial': 1, 'random': 2, 'all': 3 };

            document.querySelectorAll('.mode-btn')[btnIdx[mode]].classList.add('active');

            

            renderCard();

        }


        // 특정 번호 직접 이동

        function jumpToVerse(val) {

            let num = parseInt(val);

            if(isNaN(num) || num < 1) num = 1;

            if(num > currentFilteredList.length) num = currentFilteredList.length;

            

            currentIndex = num - 1;

            renderCard();

        }


        // 이전/다음 구절 이동

        function prevVerse() {

            if(currentIndex > 0) {

                currentIndex--;

                renderCard();

            }

        }


        function nextVerse() {

            if(currentIndex < currentFilteredList.length - 1) {

                currentIndex++;

                renderCard();

            }

        }


        // 핵심: 텍스트를 규칙에 맞게 쪼개고 가림막 처리하는 핵심 로직

        function processContent(text) {

            const words = text.split(' ');

            

            if (currentMode === 'origin') {

                return text;

            }

            

            if (currentMode === 'all') {

                return words.map(w => `<span class="blind">${w}</span>`).join(' ');

            }

            

            if (currentMode === 'partial') {

                // 한 다리 건너 하나씩 가리기 (조사 등을 고려해 징검다리 가림막)

                return words.map((w, idx) => {

                    if (idx % 2 === 1 && w.length > 1) {

                        return `<span class="blind">${w}</span>`;

                    }

                    return w;

                }).join(' ');

            }

            

            if (currentMode === 'random') {

                // 단어의 약 35%를 수학적 규칙성 없이 무작위 가리기

                return words.map(w => {

                    if (Math.random() < 0.35 && w.length > 0) {

                        return `<span class="blind">${w}</span>`;

                    }

                    return w;

                }).join(' ');

            }

            

            return text;

        }


        // 카드 컴포넌트 갱신 렌더링

        function renderCard() {

            const item = currentFilteredList[currentIndex];

            if(!item) return;


            // 메타정보 및 구절 세팅

            const categoryName = item.cat === 'OT' ? '구약' : '신약';

            document.getElementById('verseMeta').innerText = `${categoryName} ${item.num}번 (전체번호: ${item.id})`;

            document.getElementById('verseRef').innerText = `(${item.ref})`;

            

            // 본문 가림막 변환 반영

            document.getElementById('verseText').innerHTML = processContent(item.text);


            // 네비게이션 인풋 및 하단 진행바 동기화

            document.getElementById('jumpIdx').value = currentIndex + 1;

            document.getElementById('jumpIdx').max = currentFilteredList.length;

            document.getElementById('totalCount').innerText = `/ ${currentFilteredList.length}`;


            const percentage = Math.round(((currentIndex + 1) / currentFilteredList.length) * 100);

            document.getElementById('progressBar').style.width = `${percentage}%`;

            document.getElementById('progressText').innerText = `${currentIndex + 1}/${currentFilteredList.length} 구절 학습 중 (${percentage}%)`;

        }

    </script>
